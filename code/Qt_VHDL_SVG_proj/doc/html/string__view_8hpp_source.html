206 size_type copy(charT* s, size_type n, size_type pos=0) const {
207 if (pos > size())
208 SC_BOOST_THROW_EXCEPTION(std::out_of_range("string_view::copy" ));
209 size_type rlen = (std::min)(n, len_ - pos);
210 // use std::copy(begin() + pos, begin() + pos + rlen, s) rather than
211 // std::copy_n(begin() + pos, rlen, s) to support pre-C++11 standard libraries
212 std::copy(begin() + pos, begin() + pos + rlen, s);
213 return rlen;
214 }

216 SC_BOOST_CXX14_CONSTEXPR basic_string_view substr(size_type pos, size_type n=npos) const {
217 if ( pos > size())
218 SC_BOOST_THROW_EXCEPTION( std::out_of_range ( "string_view::substr" ) );
219 if (n == npos || pos + n > size())
220 n = size () - pos;
221 return basic_string_view(data() + pos, n);
222 }

224 SC_BOOST_CXX14_CONSTEXPR int compare(basic_string_view x) const SC_BOOST_NOEXCEPT {
225 const int cmp = traits::compare(ptr_, x.ptr_, (std::min)(len_, x.len_));
226 return cmp != 0 ? cmp : (len_ == x.len_ ? 0 : len_ < x.len_ ? -1 : 1);
227 }

270 // find
271 SC_BOOST_CXX14_CONSTEXPR size_type find(basic_string_view s, size_type pos = 0) const SC_BOOST_NOEXCEPT {
272 if (pos > size())
273 return npos;
274 if (s.empty())
275 return pos;
276 const_iterator iter = std::search(this->cbegin() + pos, this->cend(),
277 s.cbegin (), s.cend (), traits::eq);
278 return iter == this->cend () ? npos : std::distance(this->cbegin (), iter);
279 }
280 SC_BOOST_CXX14_CONSTEXPR size_type find(charT c, size_type pos = 0) const SC_BOOST_NOEXCEPT
281 { return find(basic_string_view(&c, 1), pos); }
282 SC_BOOST_CXX14_CONSTEXPR size_type find(const charT* s, size_type pos, size_type n) const SC_BOOST_NOEXCEPT
283 { return find(basic_string_view(s, n), pos); }
284 SC_BOOST_CXX14_CONSTEXPR size_type find(const charT* s, size_type pos = 0) const SC_BOOST_NOEXCEPT
285 { return find(basic_string_view(s), pos); }

287 // rfind
288 SC_BOOST_CXX14_CONSTEXPR size_type rfind(basic_string_view s, size_type pos = npos) const SC_BOOST_NOEXCEPT {
289 if (len_ < s.len_)
290 return npos;
291 if (pos > len_ - s.len_)
292 pos = len_ - s.len_;
293 if (s.len_ == 0u) // an empty string is always found
294 return pos;
295 for (const charT* cur = ptr_ + pos; ; --cur) {
296 if (traits::compare(cur, s.ptr_, s.len_) == 0)
297 return cur - ptr_;
298 if (cur == ptr_)
299 return npos;
300 };
301 }
302 SC_BOOST_CXX14_CONSTEXPR size_type rfind(charT c, size_type pos = npos) const SC_BOOST_NOEXCEPT
303 { return rfind(basic_string_view(&c, 1), pos); }
304 SC_BOOST_CXX14_CONSTEXPR size_type rfind(const charT* s, size_type pos, size_type n) const SC_BOOST_NOEXCEPT
305 { return rfind(basic_string_view(s, n), pos); }
306 SC_BOOST_CXX14_CONSTEXPR size_type rfind(const charT* s, size_type pos = npos) const SC_BOOST_NOEXCEPT
307 { return rfind(basic_string_view(s), pos); }

309 // find_first_of
310 SC_BOOST_CXX14_CONSTEXPR size_type find_first_of(basic_string_view s, size_type pos = 0) const SC_BOOST_NOEXCEPT {
311 if (pos >= len_ || s.len_ == 0)
312 return npos;
313 const_iterator iter = std::find_first_of
314 (this->cbegin () + pos, this->cend (), s.cbegin (), s.cend (), traits::eq);
315 return iter == this->cend () ? npos : std::distance ( this->cbegin (), iter );
316 }

318 { return find_first_of(basic_string_view(&c, 1), pos); }

320 { return find_first_of(basic_string_view(s, n), pos); }

322 { return find_first_of(basic_string_view(s), pos); }

324 // find_last_of
325 SC_BOOST_CXX14_CONSTEXPR size_type find_last_of(basic_string_view s, size_type pos = npos) const SC_BOOST_NOEXCEPT {
326 if (s.len_ == 0u)
327 return npos;
328 if (pos >= len_)
329 pos = 0;
330 else
331 pos = len_ - (pos+1);
332 const_reverse_iterator iter = std::find_first_of
333 ( this->crbegin () + pos, this->crend (), s.cbegin (), s.cend (), traits::eq );
334 return iter == this->crend () ? npos : reverse_distance ( this->crbegin (), iter);
335 }
336 SC_BOOST_CXX14_CONSTEXPR size_type find_last_of(charT c, size_type pos = npos) const SC_BOOST_NOEXCEPT
337 { return find_last_of(basic_string_view(&c, 1), pos); }
338 SC_BOOST_CXX14_CONSTEXPR size_type find_last_of(const charT* s, size_type pos, size_type n) const SC_BOOST_NOEXCEPT
339 { return find_last_of(basic_string_view(s, n), pos); }
340 SC_BOOST_CXX14_CONSTEXPR size_type find_last_of(const charT* s, size_type pos = npos) const SC_BOOST_NOEXCEPT
341 { return find_last_of(basic_string_view(s), pos); }

343 // find_first_not_of
344 SC_BOOST_CXX14_CONSTEXPR size_type find_first_not_of(basic_string_view s, size_type pos = 0) const SC_BOOST_NOEXCEPT {
345 if (pos >= len_)
346 return npos;
347 if (s.len_ == 0)
348 return pos;
349 const_iterator iter = find_not_of ( this->cbegin () + pos, this->cend (), s );
350 return iter == this->cend () ? npos : std::distance ( this->cbegin (), iter );
351 }
352 SC_BOOST_CXX14_CONSTEXPR size_type find_first_not_of(charT c, size_type pos = 0) const SC_BOOST_NOEXCEPT
353 { return find_first_not_of(basic_string_view(&c, 1), pos); }
354 SC_BOOST_CXX14_CONSTEXPR size_type find_first_not_of(const charT* s, size_type pos, size_type n) const SC_BOOST_NOEXCEPT
355 { return find_first_not_of(basic_string_view(s, n), pos); }
356 SC_BOOST_CXX14_CONSTEXPR size_type find_first_not_of(const charT* s, size_type pos = 0) const SC_BOOST_NOEXCEPT
357 { return find_first_not_of(basic_string_view(s), pos); }
358
359 // find_last_not_of
360 SC_BOOST_CXX14_CONSTEXPR size_type find_last_not_of(basic_string_view s, size_type pos = npos) const SC_BOOST_NOEXCEPT {
361 if (pos >= len_)
362 pos = len_ - 1;
363 if (s.len_ == 0u)
364 return pos;
365 pos = len_ - (pos+1);
366 const_reverse_iterator iter = find_not_of ( this->crbegin () + pos, this->crend (), s );
367 return iter == this->crend () ? npos : reverse_distance ( this->crbegin (), iter );
368 }
369 SC_BOOST_CXX14_CONSTEXPR size_type find_last_not_of(charT c, size_type pos = npos) const SC_BOOST_NOEXCEPT
370 { return find_last_not_of(basic_string_view(&c, 1), pos); }
371 SC_BOOST_CXX14_CONSTEXPR size_type find_last_not_of(const charT* s, size_type pos, size_type n) const SC_BOOST_NOEXCEPT
372 { return find_last_not_of(basic_string_view(s, n), pos); }
373 SC_BOOST_CXX14_CONSTEXPR size_type find_last_not_of(const charT* s, size_type pos = npos) const SC_BOOST_NOEXCEPT
374 { return find_last_not_of(basic_string_view(s), pos); }
375
376 private:
377 template <typename r_iter>
378 size_type reverse_distance(r_iter first, r_iter last) const SC_BOOST_NOEXCEPT {
379 // Portability note here: std::distance is not NOEXCEPT, but calling it with a string_view::reverse_iterator will not throw.
380 return len_ - 1 - std::distance ( first, last );
381 }
382
383 template <typename Iterator>
384 Iterator find_not_of(Iterator first, Iterator last, basic_string_view s) const SC_BOOST_NOEXCEPT {
385 for (; first != last ; ++first)
386 if ( 0 == traits::find(s.ptr_, s.len_, *first))
387 return first;
388 return last;
389 }
390
391 const charT *ptr_;
392 std::size_t len_;
393 };

588 template<class charT, class traits>
589 inline void sv_insert_fill_chars(std::basic_ostream<charT, traits>& os, std::size_t n) {
590 enum { chunk_size = 8 };
591 charT fill_chars[chunk_size];
592 std::fill_n(fill_chars, static_cast< std::size_t >(chunk_size), os.fill());
593 for (; n >= chunk_size && os.good(); n -= chunk_size)
594 os.write(fill_chars, static_cast< std::size_t >(chunk_size));
595 if (n > 0 && os.good())
596 os.write(fill_chars, n);
597 }
598
599 template<class charT, class traits>
600 void sv_insert_aligned(std::basic_ostream<charT, traits>& os, const basic_string_view<charT,traits>& str) {
601 const std::size_t size = str.size();
602 const std::size_t alignment_size = static_cast< std::size_t >(os.width()) - size;
603 const bool align_left = (os.flags() & std::basic_ostream<charT, traits>::adjustfield) == std::basic_ostream<charT, traits>::left;
604 if (!align_left) {
605 detail::sv_insert_fill_chars(os, alignment_size);
606 if (os.good())
607 os.write(str.data(), size);
608 }
609 else {
610 os.write(str.data(), size);
611 if (os.good())
612 detail::sv_insert_fill_chars(os, alignment_size);
613 }
614 }
