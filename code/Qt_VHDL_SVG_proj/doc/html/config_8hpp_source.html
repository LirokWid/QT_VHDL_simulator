17#ifndef SC_BOOST_CONFIG_HPP
18#define SC_BOOST_CONFIG_HPP
19
20// if we don't have a user config, then use the default location:
21#if !defined(SC_BOOST_USER_CONFIG) && !defined(SC_BOOST_NO_USER_CONFIG)
22# define SC_BOOST_USER_CONFIG <sysc/packages/boost/config/user.hpp>
23#endif
24// include it first:
25#ifdef SC_BOOST_USER_CONFIG
26# include SC_BOOST_USER_CONFIG
27#endif
28
29// if we don't have a compiler config set, try and find one:
30#if !defined(SC_BOOST_COMPILER_CONFIG) && !defined(SC_BOOST_NO_COMPILER_CONFIG) && !defined(SC_BOOST_NO_CONFIG)
31# include <sysc/packages/boost/config/select_compiler_config.hpp>
32#endif
33// if we have a compiler config, include it now:
34#ifdef SC_BOOST_COMPILER_CONFIG
35# include SC_BOOST_COMPILER_CONFIG
36#endif
37
38// if we don't have a std library config set, try and find one:
39#if !defined(SC_BOOST_STDLIB_CONFIG) && !defined(SC_BOOST_NO_STDLIB_CONFIG) && !defined(SC_BOOST_NO_CONFIG)
40# include <sysc/packages/boost/config/select_stdlib_config.hpp>
41#endif
42// if we have a std library config, include it now:
43#ifdef SC_BOOST_STDLIB_CONFIG
44# include SC_BOOST_STDLIB_CONFIG
45#endif
46
47// if we don't have a platform config set, try and find one:
48#if !defined(SC_BOOST_PLATFORM_CONFIG) && !defined(SC_BOOST_NO_PLATFORM_CONFIG) && !defined(SC_BOOST_NO_CONFIG)
49# include <sysc/packages/boost/config/select_platform_config.hpp>
50#endif
51// if we have a platform config, include it now:
52#ifdef SC_BOOST_PLATFORM_CONFIG
53# include SC_BOOST_PLATFORM_CONFIG
54#endif
55
56// get config suffix code:
57#include <sysc/packages/boost/config/suffix.hpp>
58
59#endif // SC_BOOST_CONFIG_HPP
