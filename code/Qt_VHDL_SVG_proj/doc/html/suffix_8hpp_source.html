333// SC_BOOST_USE_FACET / HAS_FACET workaround ----------------------------------//
334// When the standard library does not have a conforming std::use_facet there
335// are various workarounds available, but they differ from library to library.
336// The same problem occurs with has_facet.
337// These macros provide a consistent way to access a locale's facets.
338// Usage:
339// replace
340// std::use_facet<Type>(loc);
341// with
342// SC_BOOST_USE_FACET(Type, loc);
343// Note do not add a std:: prefix to the front of SC_BOOST_USE_FACET!
344// Use for SC_BOOST_HAS_FACET is analagous.
345
346#if defined(SC_BOOST_NO_STD_USE_FACET)
347# ifdef SC_BOOST_HAS_TWO_ARG_USE_FACET
348# define SC_BOOST_USE_FACET(Type, loc) std::use_facet(loc, static_cast<Type*>(0))
349# define SC_BOOST_HAS_FACET(Type, loc) std::has_facet(loc, static_cast<Type*>(0))
350# elif defined(SC_BOOST_HAS_MACRO_USE_FACET)
351# define SC_BOOST_USE_FACET(Type, loc) std::_USE(loc, Type)
352# define SC_BOOST_HAS_FACET(Type, loc) std::_HAS(loc, Type)
353# elif defined(SC_BOOST_HAS_STLP_USE_FACET)
354# define SC_BOOST_USE_FACET(Type, loc) (*std::_Use_facet<Type >(loc))
355# define SC_BOOST_HAS_FACET(Type, loc) std::has_facet< Type >(loc)
356# endif
357#else
358# define SC_BOOST_USE_FACET(Type, loc) std::use_facet< Type >(loc)
359# define SC_BOOST_HAS_FACET(Type, loc) std::has_facet< Type >(loc)
360#endif

423// SC_BOOST_[APPEND_]EXPLICIT_TEMPLATE_[NON_]TYPE macros --------------------------//
424//
425// Some compilers have problems with function templates whose
426// template parameters don't appear in the function parameter
427// list (basically they just link one instantiation of the
428// template in the final executable). These macros provide a
429// uniform way to cope with the problem with no effects on the
430// calling syntax.
431
432// Example:
433//
434// #include <iostream>
435// #include <ostream>
436// #include <typeinfo>
437//
438// template <int n>
439// void f() { std::cout << n << ' '; }
440//
441// template <typename T>
442// void g() { std::cout << typeid(T).name() << ' '; }
443//
444// int main() {
445// f<1>();
446// f<2>();
447//
448// g<int>();
449// g<double>();
450// }
451//
452// With VC++ 6.0 the output is:
453//
454// 2 2 double double
455//
456// To fix it, write
457//
458// template <int n>
459// void f(SC_BOOST_EXPLICIT_TEMPLATE_NON_TYPE(int, n)) { ... }
460//
461// template <typename T>
462// void g(SC_BOOST_EXPLICIT_TEMPLATE_TYPE(T)) { ... }
463//
464
465
466#if defined SC_BOOST_NO_EXPLICIT_FUNCTION_TEMPLATE_ARGUMENTS
467
468# include "sysc/packages/boost/type.hpp"
469# include "sysc/packages/boost/non_type.hpp"
470
471# define SC_BOOST_EXPLICIT_TEMPLATE_TYPE(t) sc_boost::type<t>* = 0
472# define SC_BOOST_EXPLICIT_TEMPLATE_TYPE_SPEC(t) sc_boost::type<t>*
473# define SC_BOOST_EXPLICIT_TEMPLATE_NON_TYPE(t, v) sc_boost::non_type<t, v>* = 0
474# define SC_BOOST_EXPLICIT_TEMPLATE_NON_TYPE_SPEC(t, v) sc_boost::non_type<t, v>*
475
476# define SC_BOOST_APPEND_EXPLICIT_TEMPLATE_TYPE(t) \
477 , SC_BOOST_EXPLICIT_TEMPLATE_TYPE(t)
478# define SC_BOOST_APPEND_EXPLICIT_TEMPLATE_TYPE_SPEC(t) \
479 , SC_BOOST_EXPLICIT_TEMPLATE_TYPE_SPEC(t)
480# define SC_BOOST_APPEND_EXPLICIT_TEMPLATE_NON_TYPE(t, v) \
481 , SC_BOOST_EXPLICIT_TEMPLATE_NON_TYPE(t, v)
482# define SC_BOOST_APPEND_EXPLICIT_TEMPLATE_NON_TYPE_SPEC(t, v) \
