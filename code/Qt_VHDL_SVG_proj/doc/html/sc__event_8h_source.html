57// ----------------------------------------------------------------------------
58// CLASS : sc_event_expr
59//
60// The event expression class.
61// ----------------------------------------------------------------------------
62
63template< typename T >
64class sc_event_expr
65{
66 friend class sc_event;
67 friend class sc_event_and_list;
68 friend class sc_event_or_list;
69
70 typedef T type;
71
72 inline sc_event_expr()
73 : m_expr( new T(true) )
74 {}
75
76public:
77
78 inline sc_event_expr( sc_event_expr const & e) // move semantics
79 : m_expr(e.m_expr)
80 {
81 e.m_expr = 0;
82 }
83
84 T const & release() const
85 {
86 sc_assert( m_expr );
87 T* expr = m_expr;
88 m_expr=0;
89 return *expr;
90 }
91
92 void push_back( sc_event const & e) const
93 {
94 sc_assert( m_expr );
95 m_expr->push_back(e);
96 }
97
98 void push_back( type const & el) const
99 {
100 sc_assert( m_expr );
101 m_expr->push_back(el);
102 }
103 operator T const &() const
104 {
105 return release();
106 }
107
108 ~sc_event_expr()
109 {
110 delete m_expr;
111 }
112
113private:
114 mutable type * m_expr;
115
116 // disabled
117 void operator=( sc_event_expr const & );
118};

408inline
409void
410sc_event::notify( double v, sc_time_unit tu )
411{
412 notify( sc_time( v, tu, m_simc ) );
413}
414
415
416inline
417void
418sc_event::notify_internal( const sc_time& t )
419{
420 if( t == SC_ZERO_TIME ) {
421 // add this event to the delta events set
422 m_delta_event_index = m_simc->add_delta_event( this );
423 m_notify_type = DELTA;
424 } else {
425 sc_event_timed* et =
426 new sc_event_timed( this, m_simc->time_stamp() + t );
427 m_simc->add_timed_event( et );
428 m_timed = et;
429 m_notify_type = TIMED;
430 }
431}
432
433inline
434void
435sc_event::notify_next_delta()
436{
437 if( m_notify_type != NONE ) {
438 SC_REPORT_ERROR( SC_ID_NOTIFY_DELAYED_, 0 );
439 }
440 // add this event to the delta events set
441 m_delta_event_index = m_simc->add_delta_event( this );
442 m_notify_type = DELTA;
443}
444
445inline
446void
447sc_event::notify_delayed( double v, sc_time_unit tu )
448{
449 notify_delayed( sc_time( v, tu, m_simc ) );
450}

493inline
494sc_event_list::sc_event_list( bool and_list_, bool auto_delete_ )
495 : m_events()
496 , m_and_list( and_list_ )
497 , m_auto_delete( auto_delete_ )
498 , m_busy( 0 )

500}
501
502inline
503sc_event_list::sc_event_list( const sc_event& e,
504 bool and_list_,
505 bool auto_delete_ )
506 : m_events()
507 , m_and_list( and_list_ )
508 , m_auto_delete( auto_delete_ )
509 , m_busy(0)
510{
511 m_events.push_back( &e );
512}
513
514inline
515sc_event_list::sc_event_list( sc_event_list const & that )
516 : m_events()
517 , m_and_list( that.m_and_list )
518 , m_auto_delete( false )
519 , m_busy( 0 )
520{
521 move_from( that );
522 that.auto_delete(); // free automatic lists
523}
524
525inline
526sc_event_list&
527sc_event_list::operator=( sc_event_list const & that )
528{
529 if( m_busy )
530 report_invalid_modification();
531
532 if( SC_LIKELY_(this != &that) ) {
533 move_from( that );
534 that.auto_delete(); // free automatic lists
535 }
536
537 return *this;
538}
539
540inline
541sc_event_list::~sc_event_list()
542{
543 if( m_busy )
544 report_premature_destruction();
545}
546
547inline
548void
549sc_event_list::swap( sc_event_list& that )
550{
551 if( busy() || that.busy() )
552 report_invalid_modification();
553 m_events.swap( that.m_events );
554}
555
556inline
557void
558sc_event_list::move_from( sc_event_list const& that )
559{
560 if( that.temporary() ) {
561 swap( const_cast<sc_event_list&>(that) ); // move from source
562 } else {
563 m_events = that.m_events; // copy from source
564 }
565}

590inline
591bool
592sc_event_list::temporary() const
593{
594 return m_auto_delete && ! m_busy;

596
597inline
598void
599sc_event_list::auto_delete() const
600{
601 if( m_busy ) {
602 --m_busy;
603 }
604 if( ! m_busy && m_auto_delete ) {
605 delete this;
606 }
607}
