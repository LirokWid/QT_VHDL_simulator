63// shared implementation of trace files
64class SC_API sc_trace_file_base
65 : public sc_trace_file
66#if SC_TRACING_PHASE_CALLBACKS_
67 , private sc_object // to be used as callback target
68#endif
69{
70public:
71 typedef sc_time::value_type unit_type;
72
73 const char* filename() const
74 { return filename_.c_str(); }
75
76 bool delta_cycles() const
77 { return trace_delta_cycles_; }
78
79 // Also trace transitions between delta cycles if flag is true.
80 virtual void delta_cycles(bool flag);
81
82 // set a user-define timescale unit for the trace file
83 virtual void set_time_unit( double v, sc_time_unit tu);
84
85protected:
86 sc_trace_file_base( const char* name, const char* extension );
87
88 // returns true, if trace file is already initialized
89 bool is_initialized() const;
90 // returns true, iff initialization has been performed in this call
91 bool initialize();
92 // ensure that file has been opened (needed for early write_comment())
93 void open_fp();
94 // perform format specific initialization
95 virtual void do_initialize() = 0;
96
97 // returns true, if new trace objects can still be added
98 // (i.e. trace file is not yet initialized)
99 bool add_trace_check( const std::string& name ) const;
100
101 // tracefile time unit < kernel unit, extra units will be placed in low part
102 bool has_low_units() const;
103
104 // number of decimal digits in low units
105 int low_units_len() const;
106
107 // get current kernel time in trace time units
108 void timestamp_in_trace_units(unit_type &high, unit_type &low) const;
109
110 // Flush results and close file.
111 virtual ~sc_trace_file_base();
112
113 static sc_time::value_type unit_to_fs(sc_time_unit tu);
114
115 static std::string fs_unit_to_str(sc_trace_file_base::unit_type tu);
116
117#if SC_TRACING_PHASE_CALLBACKS_
118private:
119 virtual void simulation_phase_callback();
120#endif // SC_TRACING_PHASE_CALLBACKS_
121
122protected:
123 FILE* fp; // pointer to the trace file
124
125 unit_type trace_unit_fs; // tracefile timescale unit in femtoseconds
126 unit_type kernel_unit_fs; // kernel timescale unit in femtoseconds
127 bool timescale_set_by_user; // = true means set by user
128
129private:
130 std::string filename_; // name of the file (for reporting)
131 bool initialized_; // tracing started?
132 bool trace_delta_cycles_; // also trace delta transitions?
133
134 static bool tracing_initialized_; // shared setup of tracing implementation
135
136private: // disabled
137 sc_trace_file_base( const sc_trace_file_base& ) /* = delete */;
138 sc_trace_file_base& operator=( const sc_trace_file_base& ) /* = delete */;
139
140}; // class sc_trace_file_base
