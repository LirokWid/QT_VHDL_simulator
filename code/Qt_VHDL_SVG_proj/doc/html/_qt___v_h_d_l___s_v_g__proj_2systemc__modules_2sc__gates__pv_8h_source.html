81 static bool validate(QString gate, QString id, QDomElement gElement) {
82 QDomNodeList children = gElement.elementsByTagName("id");
83 bool ninputs_ok;
84
85 // To validate the logic gate is in the catalog, only the number of inputs is required
86 // (the bits/input is irrelevant at this moment)
87 int ninputs = getNInputs(id, ninputs_ok);
88
89 // vvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvv
90 // Rework this part to accept: and, and_wide, and2, and2_wide, ...,
91 // AND2, ANDn, AND2_WIDE, ANDn_WIDE, (lowercase, uppercase, camelcase)
92 // Being n the number of inputs (D_0, D_1, ..., D_n-1)
93 // Y is the output
94 // @remark: Be very "picky" here, reject *anything that doesn't match perfectly
95 if (!id.toUpper().startsWith(gate)) {
96 return false;
97 }
98 if (!ninputs_ok) {
99 return false;
100 }
101 // ^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^
102
103 if (!find("y", children)) {
104 return false;
105 }
106 for (auto i = 0; i < ninputs; i++) {
107 // For the SyscLogicGate class and its derivatives,
108 // it should be "d" below (not "d_")
109 if (!find("d_" + QString::number(i), children)) {
110 return false;
111 }
112 }
113
114 return true;
