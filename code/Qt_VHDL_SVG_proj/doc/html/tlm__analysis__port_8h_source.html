30template < typename T>
31class tlm_analysis_port :
32 public sc_core::sc_object ,

35 public:
36 tlm_analysis_port() : sc_core::sc_object() {}
37 tlm_analysis_port( const char *nm ) : sc_core::sc_object( nm ) {}
38
39 // bind and () work for both interfaces and analysis ports, since
40 // analysis ports implement the analysis interface
41
42 virtual void bind( tlm_analysis_if<T> &_if ) {
43 m_interfaces.push_back( &_if );
44 }
45
46 void operator() ( tlm_analysis_if<T> &_if ) { bind( _if ); }
47
48 virtual bool unbind( tlm_analysis_if<T> &_if ) {
49
50 typename std::deque< tlm_analysis_if<T> * >::iterator i
51 = std::remove( m_interfaces.begin(), m_interfaces.end(), &_if );
52
53 if( i != m_interfaces.end() ) {
54 m_interfaces.erase(i, m_interfaces.end() );
55 return 1;
56 }
57
58 return 0;
59
60 }
61
62 void write( const T &t ) {
63 typename std::deque< tlm_analysis_if<T> * >::iterator i;
64
65 for( i = m_interfaces.begin();
66 i != m_interfaces.end();
67 i++ ) {
68
69 (*i)->write( t );
70
71 }
72
73 }
74
75 private:
76 std::deque< tlm_analysis_if<T> * > m_interfaces;
77
78};
